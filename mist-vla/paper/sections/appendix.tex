\mathcal{L}_{\text{fail}}&=\text{BCEWithLogits}(\hat{y}^{\text{fail}}, y; w^+_{\text{batch}}),\\
\mathcal{L}_{\tau}&=\text{Huber}_{\delta}(\hat{\tau},\tau),\\
\mathcal{L}_{\Delta p}&=\text{Huber}_{\delta}(\widehat{\Delta\mathbf{p}},\Delta\mathbf{p}).
\end{align}
