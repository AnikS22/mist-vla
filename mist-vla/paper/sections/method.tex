\section{Method}

We hypothesize that frozen robot policies encode latent safety signals---failure risk, temporal urgency, and corrective geometry readable from hidden states. To test this, we train a lightweight MLP probe that reads these signals and produces corrective steering outputs. The only assumption is access to per-step hidden states.

\subsection{Problem Formulation}
At each control step $t$, a frozen policy emits hidden state $\mathbf{h}_t \in \mathbb{R}^m$ and action $\mathbf{a}_t \in \mathbb{R}^d$. We learn a \textbf{latent safety controller}:
\begin{equation}
f_\theta(\mathbf{h}_t) = \bigl\{\hat{y}^{\mathrm{fail}}_t,\;\hat{\tau}_t,\;\widehat{\Delta\mathbf{p}}_t\bigr\},
\end{equation}
where $\hat{y}^{\mathrm{fail}}_t$ is the failure logit, $\hat{\tau}_t$ is time-to-failure, and $\widehat{\Delta\mathbf{p}}_t = (\Delta x, \Delta y, \Delta z)$ is a Cartesian correction vector.

\subsection{Architecture}
The safety probe is a compact MLP (108K params for ACT 256-d; 1.1M for OpenVLA 4096-d):
\begin{itemize}[leftmargin=1.2em, topsep=2pt, itemsep=1pt]
    \item \textbf{Input:} LayerNorm on the policy's hidden state (any dimension $m$).
    \item \textbf{Encoder:} Bottleneck: $m \to 256 \to 128 \to 64$, GELU, LayerNorm, Dropout(0.3).
    \item \textbf{Three heads:} Linear projections from 64-d to failure logit, TTF scalar, 3-D correction.
\end{itemize}

\subsection{Training}
\begin{equation}
\mathcal{L} = \lambda_f \,\mathrm{BCE}(\hat{y}^{\mathrm{fail}}, y;\; w^+) + \lambda_\tau \,\mathrm{Huber}(\hat{\tau}, \tau) + \lambda_c \,\mathrm{Huber}(\widehat{\Delta\mathbf{p}}, \Delta\mathbf{p}).
\end{equation}
Labels come from paired success/failure trajectories: $\Delta\mathbf{p}_t = \mathbf{p}^{\mathrm{succ}}_t - \mathbf{p}^{\mathrm{fail}}_t$ in Cartesian EEF space (Figure~\ref{fig:pipeline}, top). The MLP sees only hidden states; EEF positions are used only for training labels.

\subsection{Inference-Time Steering}
At deployment, corrections are smoothed, clamped, and gated:
\begin{align}
\tilde{\Delta\mathbf{p}}_t &= \beta\,\tilde{\Delta\mathbf{p}}_{t-1} + (1{-}\beta)\,\widehat{\Delta\mathbf{p}}_t, \label{eq:ema} \\
\bar{\Delta\mathbf{p}}_t &= \mathrm{ClampNorm}(\tilde{\Delta\mathbf{p}}_t,\; c_{\max}), \label{eq:clamp} \\
\mathbb{I}_t &= \mathbb{1}[\|\bar{\Delta\mathbf{p}}_t\| > \tau_c] \cdot \mathbb{1}[\sigma(\hat{y}^{\mathrm{fail}}_t) \ge \tau_f], \label{eq:gate} \\
\mathbf{a}^{xyz}_t &\leftarrow \mathbf{a}^{xyz}_t + \alpha\,\mathbb{I}_t\,\bar{\Delta\mathbf{p}}_t / s_a. \label{eq:apply}
\end{align}
\textbf{Double gating} (Eq.~\ref{eq:gate}): corrections fire only when both correction magnitude and failure probability exceed thresholds, ensuring sparse intervention ($\sim$46\% of timesteps). \textbf{Clamping} (Eq.~\ref{eq:clamp}) acts as a trust region.
