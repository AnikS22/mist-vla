\section{Experimental Setup}

\subsection{Benchmarks and Task Suites}
Primary evaluation uses LIBERO spatial tasks with off-screen rendering and fixed initialization protocols.

\subsection{Models}
\begin{itemize}[leftmargin=1.2em]
    \item OpenVLA-7B OFT (foundation policy)
    \item ACT (imitation policy)
    \item Auxiliary collected evidence: Diffusion Policy, Octo (where available)
\end{itemize}

\subsection{Evaluation Modes}
\begin{itemize}[leftmargin=1.2em]
    \item vanilla
    \item latent\_stop
    \item noise
    \item ema\_only
    \item latent\_jiggle
    \item mppi
    \item steering (ours)
\end{itemize}

\subsection{Generalization Protocols}
\paragraph{Strict zero-shot task split.}
Train safety head on task subset A, evaluate on disjoint unseen subset B.

\paragraph{OOD stress testing.}
Use disturbance OOD and/or custom OOD BDDL environments.

\subsection{Metrics}
Success rate, $\Delta$ vs vanilla, $\Delta$(Ours$-$MPPI), intervention rate, correction magnitude, and controller apply latency (mean/p95).
